 49:     /**
 50:      * Creates a new binding.
 51:      *
 52:      * A binding has a query that is used to retrieve the resources matched
 53:      * by the binding.
 54:      *
 55:      * You can pass parameters that have been defined for the type. If you pass
 56:      * unknown parameters, or if a required parameter is missing, an exception
 57:      * is thrown.
 58:      *
 59:      * All parameters that you do not set here will receive the default values
 60:      * set for the parameter.
 61:      *
 62:      * @param string      $query           The resource query.
 63:      * @param BindingType $type            The type to bind against.
 64:      * @param array       $parameterValues The values of the parameters defined
 65:      *                                     for the type.
 66:      * @param string      $language        The language of the resource query.
 67:      *
 68:      * @throws NoSuchParameterException If an invalid parameter was passed.
 69:      * @throws MissingParameterException If a required parameter was not passed.
 70:      */

 72:     {

 74:

 76:

 79:         $this->query = $query;
 80:         $this->language = $language;
 81:         $this->type = $type;

 83:     }

145: 
146:         if ($this->query !== $other->getQuery()) {

151:             return false;

157: 
158:         // The local parameters are sorted by key. Sort before comparing to

170:     private function assertParametersValid(array $parameterValues, BindingType $type)
171:     {
172:         $validator = new SimpleParameterValidator();
173:         $violations = $validator->validate($parameterValues, $type);
174: 
175:         foreach ($violations as $violation) {
176:             switch ($violation->getCode()) {
177:                 case ConstraintViolation::NO_SUCH_PARAMETER:
178:                     throw NoSuchParameterException::forParameterName($violation->getParameterName(), $violation->getTypeName());
179:                 case ConstraintViolation::MISSING_PARAMETER:
180:                     throw MissingParameterException::forParameterName($violation->getParameterName(), $violation->getTypeName());
181:             }
182:         }
183:     }
